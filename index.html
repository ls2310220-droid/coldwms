// ============================================================ // ColdWMS — Google Apps Script Backend // Cole este código em: Extensões > Apps Script > Colar > Salvar // Depois clique em "Implantar > Nova implantação > App da Web" // Acesso: Qualquer pessoa | Execute como: Você // ============================================================ const SHEET_RECEPCAO = 'Recepção'; const SHEET_ESTOQUE = 'Estoque'; const SHEET_PREFRIO = 'Pré-Frio'; const SHEET_EXPEDICAO = 'Expedição'; // ── Inicializa abas se não existirem ────────────────────────── function setup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const abas = [ { nome: SHEET_RECEPCAO, cabecalho: ['ID Pallet','SKU','Linha Origem','Caixas','Temperatura','Destino','Avaria','Obs','Turno','Hora','Data','Timestamp','Operador','Alerta Temp'] }, { nome: SHEET_ESTOQUE, cabecalho: ['Tipo','ID Pallet','SKU','Caixas','Posição','Origem','Obs','Turno','Hora','Data','Timestamp','Operador'] }, { nome: SHEET_PREFRIO, cabecalho: ['ID Pallet','Temperatura','Duração','Mercado','Observação','Hora','Data','Timestamp'] }, { nome: SHEET_EXPEDICAO, cabecalho: ['Pedido','ID Pallet','SKU','Caixas','Cliente','Placa','Motorista','Status','Hora','Data','Timestamp'] }, ]; abas.forEach(({ nome, cabecalho }) => { let aba = ss.getSheetByName(nome); if (!aba) { aba = ss.insertSheet(nome); aba.appendRow(cabecalho); // Formata cabeçalho const hdr = aba.getRange(1, 1, 1, cabecalho.length); hdr.setBackground('#0e1624').setFontColor('#00c6ff') .setFontWeight('bold').setFontFamily('Courier New'); aba.setFrozenRows(1); } }); SpreadsheetApp.getUi().alert('✅ ColdWMS configurado com sucesso!'); } // ── CORS + roteador principal ───────────────────────────────── function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const headers = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST', 'Access-Control-Allow-Headers': 'Content-Type', 'Content-Type': 'application/json' }; try { const params = e.parameter || {}; const body = e.postData ? JSON.parse(e.postData.contents || '{}') : {}; const action = params.action || body.action; const sheet = params.sheet || body.sheet; let result; switch (action) { case 'ping': result = { ok: true, ts: new Date().toISOString() }; break; case 'inserir': result = inserir(sheet, body.dados); break; case 'listar': result = listar(sheet, params.limit); break; case 'dashboard': result = getDashboard(); break; default: result = { erro: 'Ação desconhecida: ' + action }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ erro: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── INSERIR linha em uma aba ────────────────────────────────── function inserir(nomAba, dados) { if (!nomAba || !dados) throw new Error('sheet e dados são obrigatórios'); const ss = SpreadsheetApp.getActiveSpreadsheet(); const aba = ss.getSheetByName(nomAba); if (!aba) throw new Error('Aba não encontrada: ' + nomAba); const ts = new Date().toISOString(); const data = new Date().toLocaleDateString('pt-BR'); const hora = new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' }); let linha; switch (nomAba) { case SHEET_RECEPCAO: linha = [ dados.id, dados.sku, dados.linha, dados.caixas, dados.temp, dados.destino, dados.avaria, dados.obs, dados.turno, hora, data, ts, dados.operador || 'Sistema', dados.tempAlerta ? 'SIM' : 'NÃO' ]; break; case SHEET_ESTOQUE: linha = [ dados.tipo, dados.id, dados.sku, dados.caixas, dados.posicao, dados.origem, dados.obs, dados.turno, hora, data, ts, dados.operador || 'Sistema' ]; break; case SHEET_PREFRIO: linha = [ dados.id, dados.temp, dados.duracao, dados.mercado, dados.obs, hora, data, ts ]; break; case SHEET_EXPEDICAO: linha = [ dados.pedido, dados.id, dados.sku, dados.caixas, dados.cliente, dados.placa, dados.motorista, dados.status, hora, data, ts ]; break; default: throw new Error('Aba desconhecida: ' + nomAba); } aba.appendRow(linha); // Colorir linha de alerta de temperatura if (nomAba === SHEET_RECEPCAO && dados.tempAlerta) { const lastRow = aba.getLastRow(); aba.getRange(lastRow, 1, 1, linha.length) .setBackground('#2d0a0f').setFontColor('#ff4d6d'); } return { ok: true, linha: linha.length, sheet: nomAba }; } // ── LISTAR últimas N linhas ─────────────────────────────────── function listar(nomAba, limit) { if (!nomAba) throw new Error('sheet é obrigatório'); const ss = SpreadsheetApp.getActiveSpreadsheet(); const aba = ss.getSheetByName(nomAba); if (!aba) throw new Error('Aba não encontrada: ' + nomAba); const n = parseInt(limit) || 200; const lastRow = aba.getLastRow(); const headers = aba.getRange(1, 1, 1, aba.getLastColumn()).getValues()[0]; if (lastRow <= 1) return { ok: true, dados: [], headers }; const startRow = Math.max(2, lastRow - n + 1); const numRows = lastRow - startRow + 1; const values = aba.getRange(startRow, 1, numRows, aba.getLastColumn()).getValues(); const dados = values.reverse().map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return { ok: true, dados, headers, total: lastRow - 1 }; } // ── DASHBOARD — agregados ───────────────────────────────────── function getDashboard() { const ss = SpreadsheetApp.getActiveSpreadsheet(); function contarAba(nome) { const aba = ss.getSheetByName(nome); return aba ? Math.max(0, aba.getLastRow() - 1) : 0; } // Recepção: pallets de hoje const abaRec = ss.getSheetByName(SHEET_RECEPCAO); const hoje = new Date().toLocaleDateString('pt-BR'); let palletsHoje = 0, avariasHoje = 0, tempSoma = 0, tempCount = 0; let manha = 0, tarde = 0, noite = 0; if (abaRec && abaRec.getLastRow() > 1) { const dados = abaRec.getRange(2, 1, abaRec.getLastRow() - 1, abaRec.getLastColumn()).getValues(); dados.forEach(row => { if (row[10] === hoje) { // col Data palletsHoje++; if (['parcial','total','temp'].includes(row[6])) avariasHoje++; if (row[4] !== '' && !isNaN(row[4])) { tempSoma += parseFloat(row[4]); tempCount++; } if (row[8] === 'Manhã') manha++; else if (row[8] === 'Tarde') tarde++; else if (row[8] === 'Noite') noite++; } }); } // Estoque: saldo const abaEst = ss.getSheetByName(SHEET_ESTOQUE); let saldoPallets = 0; if (abaEst && abaEst.getLastRow() > 1) { const dados = abaEst.getRange(2, 1, abaEst.getLastRow() - 1, 1).getValues(); dados.forEach(row => { if (row[0] === 'entrada') saldoPallets++; else if (row[0] === 'saida') saldoPallets--; }); } return { ok: true, recepcao: { palletsHoje, avariasHoje, tempMedia: tempCount ? (tempSoma / tempCount).toFixed(1) : null, turno: { manha, tarde, noite } }, estoque: { saldo: saldoPallets }, totais: { recepcao: contarAba(SHEET_RECEPCAO), estoque: contarAba(SHEET_ESTOQUE), expedicao: contarAba(SHEET_EXPEDICAO), } }; }
